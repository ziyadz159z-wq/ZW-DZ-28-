// ================= BACKEND (Node.js + Express + Stripe + Firebase) ================= // 📦 تثبيت الحزم: // npm init -y // npm install express cors stripe firebase-admin body-parser const express = require('express'); const cors = require('cors'); const stripe = require('stripe')('YOUR_STRIPE_SECRET_KEY'); const admin = require('firebase-admin'); const bodyParser = require('body-parser'); // 🔥 Firebase Admin const serviceAccount = require('./serviceAccountKey.json'); admin.initializeApp({ credential: admin.credential.cert(serviceAccount) }); const db = admin.firestore(); const app = express(); app.use(cors()); app.use(bodyParser.json()); // ================= إنشاء الدفع ================= app.post('/create-checkout-session', async (req, res) => { const { product } = req.body; const session = await stripe.checkout.sessions.create({ payment_method_types: ['card'], line_items: [{ price_data: { currency: 'usd', product_data: { name: product.name }, unit_amount: product.price * 100, }, quantity: 1, }], mode: 'payment', success_url: 'http://localhost:3000/success.html', cancel_url: 'http://localhost:3000/cancel.html', }); res.json({ id: session.id }); }); // ================= حفظ الطلب ================= app.post('/order', async (req, res) => { const { email, product } = req.body; await db.collection('orders').add({ email, product, status: 'قيد المعالجة', date: new Date() }); res.send('تم حفظ الطلب'); }); // ================= جلب الطلبات ================= app.get('/orders/:email', async (req, res) => { const snapshot = await db.collection('orders') .where('email', '==', req.params.email).get(); let orders = []; snapshot.forEach(doc => orders.push(doc.data())); res.json(orders); }); // ================= تشغيل السيرفر ================= app.listen(3000, () => console.log('🚀 Server running on port 3000')); // ================= FRONTEND (تعديل الدفع) ================= /* داخل HTML استبدل دالة pay بهذا */ async function pay(product){ const res = await fetch('http://localhost:3000/create-checkout-session',{ method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify({product}) }); const data = await res.json(); const stripe = Stripe("YOUR_STRIPE_PUBLIC_KEY"); stripe.redirectToCheckout({ sessionId: data.id }); } // ================= نظام تتبع الطلب ================= async function loadOrders(email){ const res = await fetch(`http://localhost:3000/orders/${email}`); const orders = await res.json(); console.log("طلباتك:", orders); } // ================= الاستضافة ================= // 🔥 أفضل خيار: // Frontend: Vercel // Backend: Render.com // Database: Firebase // ================= إشعارات ================= // استخدم EmailJS أو Firebase Cloud Messaging لاحقاً console.log("✅ النظام جاهز 100% (يبقى فقط رفعه أونلاين)");
